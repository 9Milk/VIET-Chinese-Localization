
VIET_gift_baklava = {
	category = interaction_category_friendly
	common_interaction = yes

	}
	ai_targets = {
		ai_recipients = liege
	}
	ai_targets = {
		ai_recipients = neighboring_rulers

		max = 10
	}
	ai_target_quick_trigger = {
		adult = yes
	}
	ai_frequency = 24
	
	cooldown = { years = 1 }
	
	is_shown = {
		NOT = {
			scope:recipient = scope:actor
		}
		scope:actor = {	
			has_character_modifier = RICE_tuyuhun_character_owns_qinghaicong_horses

	}

			trigger_event = VIETmisc.2039

			}
		}
		scope:recipient = {
			show_as_tooltip = {
				custom_tooltip = VIET_decision_traveling_baklava_merchant_effect_tooltip

					target = scope:actor
					modifier = VIET_opinion_gifted_baklava

			}
